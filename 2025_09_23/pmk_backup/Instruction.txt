CREATE TABLE IF NOT EXISTS completed_trade (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            signal TEXT,                  # BUY OR SELL
            spot_entry REAL,              # INDEX PRICE
            option_symbol TEXT,           # OPTION SYMBOL
            strike INTEGER,               # OPTION STRIKE PRICE 
            expiry TEXT,                  # OPTION EXPIRY DATE
            option_sell_price REAL,       # OPTION SELLING PRICE
            entry_time TEXT,              # OPTION ENTRY TIME 
            spot_exit REAL,               # INDEX PRICE
            option_buy_price REAL,        # OPTION BUYING PRICE
            exit_time TEXT,               # OPTION EXIT TIME
            pnl REAL,                     # TRADE PROFIT PER QUANTITY
            qty INTEGER,                  # TRADED QUANTITY
            interval TEXT,               # Interval like '5minute', '15minute'
            real_trade TEXT,             # 'YES' or 'NO' to indicate real or paper trade
            entry_reason TEXT,            # Reason for entering the trade e.g., 'SIGNAL_GENERATED', 'ROLLOVER', 'MANUAL_ENTRY'
            exit_reason TEXT             # Reason for exiting the trade   e.g.,'SIGNAL_GENERATED', 'TARGET_HIT',  'MANUAL_EXIT'
        )
